import { motion } from "framer-motion";

export default function OnboardingPath() {
  const router = useRouter();

    setRole(role);

    // TODO: Save in database

      case "provider":
        router.push("/onboarding/provider");
        break;

        router.push("/dashboard/nodeholder");

        router.push("/dashboard/architect");
        break;

import { redirect } from "next/navigation";
import { getCurrentUserRole } from "@/lib/role";

export default async function Dashboard() {
  const role = await getCurrentUserRole();

  if (!role) redirect("/onboarding/path");

  switch (role) {
    case "consumer":
      redirect("/dashboard/consumer");
    case "provider":
      redirect("/dashboard/provider");
    case "nodeholder":
      redirect("/dashboard/nodeholder");
    case "architect":
      redirect("/dashboard/architect");
  }
}
